\section{Introduction \& Research Background}
% ============================================================

\begin{frame}{Research Background}
    \begin{block}{Historical Origins}
        \begin{itemize}
            \item 1959: Cocconi \& Morrison published the foundational paper in \textit{Nature}, systematically arguing that interstellar communication signals could be observable
            \item 1960: Frank Drake launched Project Ozma at Green Bank Observatory --- the birth of modern SETI
            \item Evolution from narrow-band listening, Cold War-era grand designs, to contemporary technosignature search frameworks
        \end{itemize}
    \end{block}

    \begin{block}{Core Scientific Difficulties}
        \begin{itemize}
            \item Extremely large search space with highly uncertain signal forms
            \item Chronic observational coverage gaps --- null results difficult to constrain
            \item Involves extremely rare targets,超大 parameter spaces, and frontier methodology questions
            \item The advent of SKA provides a new entry point for re-evaluating SETI's scientific significance
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Research Significance \& Core Question}
    \defination{Central Thesis}{
        How can SETI transform from a controversial exploratory topic into a testable observational science?
    }

    \begin{block}{Approach}
        \begin{itemize}
            \item Treat sky coverage, frequency coverage, and sensitivity as three critical dimensions
            \item Establish a unified observational coordinate system across historical stages
            \item Demonstrate why SKA may serve as the \textbf{facility inflection point} for new-era SETI
            \item Push SETI toward observational science characterized by parameter space constraints
        \end{itemize}
    \end{block}
\end{frame}

\section{Methodology \& Three-Dimensional Framework}
% ============================================================

\begin{frame}{Data Sources \& Literature Foundation}
    \begin{block}{Four Types of Materials}
        \begin{itemize}
            \item \textbf{ADS database}: 1960--2025 SETI publications for staged evolution analysis
            \item \textbf{Classical literature}: from Cocconi-Morrison to the technosignature discourse
            \item \textbf{Representative observations}: Breakthrough Listen's 1327-star search over 1.10--3.45 GHz
            \item \textbf{SKAO official documents}: system specifications and performance parameters
        \end{itemize}
    \end{block}

    \begin{block}{Dual Bibliographic Datasets}
        \begin{itemize}
            \item \textbf{Strict core set}: titles/abstracts containing SETI, technosignature keywords $\rightarrow$ \textbf{778 samples} (peer-reviewed astronomy)
            \item \textbf{Comprehensive set}: expanded to astronomy \& physics $\rightarrow$ \textbf{1,519 samples} for broader trends
            \item Dual approach avoids both noise from broad keywords and underestimation from overly strict boundaries
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Three-Dimensional Analysis Framework}
    \begin{block}{1. Sky Coverage --- Cumulative observed sky area (deg$^2$)}
        \begin{itemize}
            \item Targeted monitoring: approximate by $N_{\text{tgt}} \times \Omega_b$ (primary beam area)
            \item Survey programs: use publicly reported area orders of magnitude
            \item Emphasis on order-of-magnitude comparison
        \end{itemize}
    \end{block}

    \begin{block}{2. Frequency Coverage --- Actual searched frequency range (GHz)}
        \begin{itemize}
            \item Distinguish nominal instrument capability from actual program search range
            \item $\Delta f = f_{\max} - f_{\min}$: what was \emph{actually} searched, not what the instrument \emph{could} cover
        \end{itemize}
    \end{block}

    \begin{block}{3. Sensitivity --- Minimum detectable signal capability}
        \begin{itemize}
            \item Strictly uses $A_{\text{eff}}/T_{\text{sys}}$, SEFD, and radiometer equation
            \item Collecting area (m$^2$) used as intuitive proxy in visualizations
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Why Radio Has Long Dominated SETI}
    \begin{block}{Physical Advantages}
        \begin{itemize}
            \item Lower photon energy at radio frequencies $\rightarrow$ efficient for long-range information transmission
            \item Minimal atmospheric attenuation at radio wavelengths
            \item Interstellar medium attenuation relatively controlled
        \end{itemize}
    \end{block}

    \begin{block}{Technical Advantages}
        \begin{itemize}
            \item Mature pipeline: narrow-band detection, Doppler correction, RFI rejection, repeat observation
            \item High equipment overlap with conventional radio astronomy --- enables ``commensal'' (shared) observing
            \item Drake Equation provides a quantitative framework
        \end{itemize}
    \end{block}

    \defination{Key Insight}{
        Radio SETI's longevity is not just preference --- it is supported by physics, engineering maturity, and institutional ``ride-along'' feasibility.
    }
\end{frame}

\section{SETI vs.\ Traditional Observational Science}
% ============================================================

\begin{frame}{Paradigm Differences: SETI vs.\ Traditional Astronomy}
    \begin{table}
        \centering\small
        \begin{tabular}{lp{3.5cm}p{3.5cm}}
            \hline
            \textbf{Dimension} & \textbf{Traditional Astronomy} & \textbf{SETI} \\
            \hline
            Target & Objects known to exist & Existence itself uncertain \\
            Null result & Can directly constrain model parameters & Only excludes local hypotheses \\
            Positive threshold & Repeated observation + model fitting & Must exclude RFI, instrument artifacts, all natural processes \\
            Evaluation & Precision \& sample size & Public focuses on ``discovery or not'' \\
            Funding & Embedded in stable observational systems & Relies on special initiatives \\
            \hline
        \end{tabular}
    \end{table}

    \begin{block}{Cosmology as a Reference}
        Cosmology transformed from ``speculative'' to ``precision science'' through observational advances (Hubble, COBE, Planck). SETI needs the same: turning vague null results into clear parameter-space upper bounds.
    \end{block}
\end{frame}

\section{SKA's Three-Dimensional Capability Leap}
% ============================================================

\begin{frame}{Frequency: From Limited Windows to Facility-Level Wideband}
    \begin{center}
        \includegraphics[width=0.75\textwidth]{fig/fig_06_p28.png}
    \end{center}
    \vspace{-0.3em}
    {\small Figure 5.1: Frequency span comparison across representative radio SETI programs}

    \begin{block}{Order-of-Magnitude Leap}
        \begin{itemize}
            \item Ozma: $\sim$1.42 GHz single frequency; Phoenix: 1--3 GHz; BL: 1.10--3.45 GHz
            \item SKA-Low: 50--350 MHz + SKA-Mid: 0.35--15.4 GHz = facility-level wideband coverage
            \item Reduces prior bias toward ``water hole'' band; extends search to lower \& higher frequencies
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Sky Area: From Targeted Monitoring to Cumulative Coverage}
    \begin{center}
        \includegraphics[width=0.75\textwidth]{fig/fig_07_p29.png}
    \end{center}
    \vspace{-0.3em}
    {\small Figure 5.2: Sky coverage comparison (log scale)}

    \begin{block}{Structural Upgrade}
        \begin{itemize}
            \item Ozma: $\sim$0.31 deg$^2$; Phoenix: $\sim$12.8 deg$^2$; BL: $\sim$16.0 deg$^2$
            \item SKA: projected $10^3$--$10^4$ deg$^2$ cumulative potential via commensal surveys
            \item First time deep-search and wide-survey strategies coexist in one facility
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Sensitivity: Beyond ``Larger Aperture''}
    \begin{center}
        \includegraphics[width=0.75\textwidth]{fig/fig_08_p32.png}
    \end{center}
    \vspace{-0.3em}
    {\small Figure 5.3: Sensitivity proxy comparison using collecting area (log scale)}

    \begin{block}{Two Key Equations}
        \begin{itemize}
            \item Radiometer equation: $S_{\min} \approx \dfrac{(S/N)_{\min} \cdot \text{SEFD}}{\sqrt{n_p \, \Delta\nu_{\text{ch}} \, t_{\text{int}}}}$
            \item EIRP constraint: $\text{EIRP}_{\min} \approx 4\pi d^2 \, S_{\min} \, \Delta\nu_{\text{sig}}$
            \item SKA1 total collecting area $\sim 4.33 \times 10^5$ m$^2$ --- 1--2 orders higher than any single dish
        \end{itemize}
    \end{block}
\end{frame}

\section{Conclusions \& Future Perspectives}
% ============================================================

\begin{frame}{Main Conclusions}
    \begin{block}{1. Literature Dynamics}
        SETI's rise and fall are driven by \textbf{equipment capabilities, funding structures, and disciplinary context} --- not by whether signals are found.
    \end{block}

    \begin{block}{2. Three-Dimensional Framework}
        Sky coverage $\times$ frequency coverage $\times$ sensitivity reveals \textbf{structural differences} between historical programs and SKA.
    \end{block}

    \begin{block}{3. SKA's Core Significance}
        SKA's importance lies not in guaranteeing discovery, but in \textbf{reshaping the evidence base} --- pushing SETI toward parameter-space-constrained observational science.
    \end{block}

    \begin{block}{4. Paradigm Transformation}
        First simultaneous expansion of three key capabilities; null results become \textbf{meaningful upper bounds} rather than ``nothing happened.''
    \end{block}
\end{frame}

\begin{frame}{Future Perspectives}
    \begin{block}{Commensal Observation}
        SETI backends operate in parallel with regular astronomy --- no dedicated telescope time needed.
    \end{block}

    \begin{block}{Multi-Band Coordination}
        Radio as primary axis; optical/IR/UV as cross-validation for technosignature searches.
    \end{block}

    \begin{block}{Algorithm \& Machine Learning}
        Critical for RFI rejection, candidate screening, and verification at scale.
    \end{block}

    \begin{block}{Formalizing Null Results}
        Treat ``no detection'' as formal scientific output with standardized parameter-space reporting.
    \end{block}

    \begin{block}{Exoplanet \& Astrobiology Synergy}
        Coordinate target selection with exoplanet habitability studies for more physically motivated searches.
    \end{block}
\end{frame}
